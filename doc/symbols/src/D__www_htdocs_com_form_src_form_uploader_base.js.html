1 /**
  2  * @fileoverview 异步文件上传组件
  3  * @author 剑平（明河）<minghe36@126.com>,紫英<daxingplay@gmail.com>
  4  **/
  5 KISSY.add('form/uploader/base', function (S, Base, Node, UrlsInput, IframeType, AjaxType, FlashType) {
  6     var EMPTY = '', $ = Node.all, LOG_PREFIX = '[uploader]:';
  7 
  8     /**
  9      * @name Uploader
 10      * @class 异步文件上传组件，目前是使用ajax+iframe的方案，日后会加入flash方案
 11      * @constructor
 12      * @extends Base
 13      * @requires Node,UrlsInput,IframeType,AjaxType
 14      */
 15     function Uploader(config) {
 16         var self = this;
 17         //调用父类构造函数
 18         Uploader.superclass.constructor.call(self, config);
 19     }
 20 
 21     S.mix(Uploader, /** @lends Uploader*/{
 22         /**
 23          * 上传方式
 24          */
 25         type:{AUTO:'auto', IFRAME:'iframe', AJAX:'ajax', FLASH:'flash'},
 26         /**
 27          * 事件
 28          */
 29         event:{
 30             //运行
 31             RENDER:'render',
 32             //选择完文件后触发
 33             SELECT:'select',
 34             //开始上传后触发
 35             START:'start',
 36             //正在上传中时触发
 37             PROGRESS : 'progress',
 38             //上传完成（在上传成功或上传失败后都会触发）
 39             COMPLETE:'complete',
 40             //上传成功后触发
 41             SUCCESS:'success',
 42             //批量上传结束后触发
 43             UPLOAD_FILES:'uploadFiles',
 44             //取消上传后触发
 45             CANCEL:'cancel',
 46             //上传失败后触发
 47             ERROR:'error'
 48         },
 49         /**
 50          * 文件上传状态
 51          */
 52         status:{}
 53     });
 54     //继承于Base，属性getter和setter委托于Base处理
 55     S.extend(Uploader, Base, /** @lends Uploader.prototype*/{
 56         /**
 57          * 运行
 58          * @return {Uploader}
 59          */
 60         render:function () {
 61             var self = this, serverConfig = self.get('serverConfig'),
 62                 type = self.get('type'),
 63                 UploadType = self.getUploadType(type), uploadType,
 64                 uploaderTypeEvent = UploadType.event,
 65                 button;
 66             if (!UploadType) return false;
 67             //路径input实例
 68             self.set('urlsInput', self._renderUrlsInput());
 69             self._renderQueue();
 70             button = self._renderButton();
 71             //如果是flash异步上传方案，增加swfUploader的实例作为参数
 72             if (self.get('type') == Uploader.type.FLASH) {
 73                 S.mix(serverConfig, {swfUploader:button.get('swfUploader')});
 74             }
 75             //实例化上传方式类
 76             uploadType = new UploadType(serverConfig);
 77             //监听上传器上传完成事件
 78             uploadType.on(uploaderTypeEvent.SUCCESS, self._uploadCompleteHanlder, self);
 79             //监听上传器上传进度事件
 80             if (uploaderTypeEvent.PROGRESS) uploadType.on(uploaderTypeEvent.PROGRESS, self._uploadProgressHandler, self);
 81             //监听上传器上传停止事件
 82             uploadType.on(uploaderTypeEvent.STOP, self._uploadStopHanlder, self);
 83             self.set('uploadType', uploadType);
 84             self.fire(Uploader.event.RENDER);
 85             return self;
 86         },
 87         /**
 88          * 上传指定队列索引的文件
 89          * @param {Number} index 文件对应的在上传队列数组内的索引值
 90          */
 91         upload:function (index) {
 92             if (!S.isNumber(index)) return false;
 93             var self = this, uploadType = self.get('uploadType'),
 94                 queue = self.get('queue'),
 95                 file = queue.get('files')[index],
 96                 uploadParam;
 97             if (!S.isPlainObject(file)) {
 98                 S.log(LOG_PREFIX + '队列中不存在id为' + index + '的文件');
 99                 return false;
100             }
101             //如果有文件正在上传，予以阻止上传
102             if (self.get('curUploadIndex') != EMPTY) {
103                 alert('第' + self.get('curUploadIndex') + '文件正在上传，请上传完后再操作！');
104                 return false;
105             }
106             //文件上传域，如果是flash上传,input为文件数据对象
107             uploadParam = file.input.id || file.input;
108             //触发文件上传前事件
109             self.fire(Uploader.event.START, {index:index, file:file});
110             //阻止文件上传
111             if (!self.get('isAllowUpload')) return false;
112             //设置当前上传的文件id
113             self.set('curUploadIndex', index);
114             //改变文件上传状态为start
115             queue.fileStatus(index, Uploader.status.START);
116 
117             //开始上传
118             uploadType.upload(uploadParam);
119         },
120         /**
121          * 取消当前正在上传的文件的上传
122          * @param {Number} index 队列数组索引
123          * @return {Uploader}
124          */
125         cancel:function (index) {
126             var self = this, uploadType = self.get('uploadType'),
127                 queue = self.get('queue'),
128                 statuses = Uploader.status,
129                 status = queue.fileStatus(index);
130             if(S.isNumber(index) && status != statuses.SUCCESS){
131                 queue.fileStatus(index,statuses.CANCEL);
132             }else{
133                 //取消上传后刷新状态，更改路径等操作请看_uploadStopHanlder()
134                 uploadType.stop();
135                 //存在批量上传操作，继续上传其他文件
136                 self._continueUpload();
137             }
138             return self;
139         },
140         /**
141          * 停止上传动作
142          * @return {Uploader}
143          */
144         stop : function(){
145             var self = this;
146             self.set('uploadFilesStatus',EMPTY);
147             self.cancel();
148             return self;
149         },
150         /**
151          * 批量上传队列中的指定状态下的文件
152          * @param {String} status 文件上传状态名
153          * @return {Uploader}
154          */
155         uploadFiles:function (status) {
156             var self = this;
157             if (!S.isString(status)) status = Uploader.status.WAITING;
158             self.set('uploadFilesStatus', status);
159             self._uploaderStatusFile(status);
160             return self;
161         },
162         /**
163          * 上传队列中的指定状态下的文件
164          * @param {String} status 文件上传状态名
165          * @return {Uploader}
166          */
167         _uploaderStatusFile:function (status) {
168             var self = this, queue = self.get('queue'),
169                 fileIndexs = queue.getIndexs(status);
170             //没有存在需要上传的文件，退出上传
171             if (!fileIndexs.length) {
172                 self.set('uploadFilesStatus', EMPTY);
173                 self.fire(Uploader.event.UPLOAD_FILES);
174                 return false;
175             }
176             //开始上传等待中的文件
177             self.upload(fileIndexs[0]);
178             return self;
179         },
180         /**
181          * 是否支持ajax方案上传
182          * @return {Boolean}
183          */
184         isSupportAjax:function () {
185             return S.isObject(FormData);
186         },
187         /**
188          * 是否支持flash方案上传
189          * @return {Boolean}
190          */
191         isSupportFlash:function () {
192             var fpv = S.UA.fpv();
193             return S.isArray(fpv) && fpv.length > 0;
194         },
195         /**
196          * 获取上传方式类（共有iframe、ajax、flash三种方式）
197          * @return {IframeType|AjaxType|FlashType}
198          */
199         getUploadType:function (type) {
200             var self = this, types = Uploader.type,
201                 UploadType;
202             //如果type参数为auto，那么type=['ajax','flash','iframe']
203             if (type == types.AUTO) type = [types.AJAX, types.IFRAME];
204             //如果是数组，遍历获取浏览器支持的上传方式
205             if (S.isArray(type) && type.length > 0) {
206                 S.each(type, function (t) {
207                     UploadType = self._getType(t);
208                     if (UploadType) return false;
209                 });
210             } else {
211                 UploadType = self._getType(type);
212             }
213             return UploadType;
214         },
215         /**
216          * 获取上传方式
217          * @param {String} type 上传方式（根据type返回对应的上传类，比如iframe返回IframeType）
218          */
219         _getType:function (type) {
220             var self = this, types = Uploader.type, UploadType,
221                 isSupportAjax = self.isSupportAjax(),
222                 isSupportFlash = self.isSupportFlash();
223             switch (type) {
224                 case types.IFRAME :
225                     UploadType = IframeType;
226                     break;
227                 case types.AJAX :
228                     UploadType = isSupportAjax && AjaxType || false;
229                     break;
230                 case types.FLASH :
231                     UploadType = isSupportFlash && FlashType || false;
232                     break;
233                 default :
234                     S.log(LOG_PREFIX + 'type参数不合法');
235                     return false;
236             }
237             self.set('type', type);
238             return UploadType;
239         },
240         /**
241          * 运行Button上传按钮组件
242          * @return {Button}
243          */
244         _renderButton:function () {
245             var self = this, button = self.get('button');
246             if (!S.isObject(button)) {
247                 S.log(LOG_PREFIX + 'button参数不合法！');
248                 return false;
249             }
250             //监听按钮改变事件
251             button.on('change', self._select, self);
252             //运行按钮实例
253             button.render();
254             return button;
255         },
256         /**
257          * 运行Queue队列组件
258          * @return {Queue} 队列实例
259          */
260         _renderQueue:function () {
261             var self = this, queue = self.get('queue'),
262                 urlsInput = self.get('urlsInput');
263             if (!S.isObject(queue)) {
264                 S.log(LOG_PREFIX + 'queue参数不合法');
265                 return false;
266             }
267             //将上传组件实例传给队列，方便队列内部执行取消、重新上传的操作
268             queue.set('uploader', self);
269             //监听队列的删除事件
270             queue.on(queue.constructor.event.REMOVE, function (ev) {
271                 //删除该文件路径，sUrl为服务器端返回的文件路径，而url是客服端文件路径
272                 urlsInput.remove(ev.file.sUrl);
273             });
274             queue.render();
275             Uploader.status = queue.constructor.status;
276             return queue;
277         },
278         /**
279          * 选择完文件后
280          * @param {Object} ev 事件对象
281          */
282         _select:function (ev) {
283             var self = this, autoUpload = self.get('autoUpload'),
284                 queue = self.get('queue'),
285                 curId = self.get('curUploadIndex'),
286                 files = ev.files;
287             S.each(files, function (file) {
288                 //文件大小，IE浏览器下不存在
289                 if (!file.size) file.size = 0;
290                 //chrome文件名属性名为fileName，而firefox为name
291                 if (!file.name) file.name = file.fileName || EMPTY;
292                 //如果是flash上传，并不存在文件上传域input
293                 file.input = ev.input || file;
294             });
295             self.fire(Uploader.event.SELECT, {files : files});
296             //阻止文件上传
297             if (!self.get('isAllowUpload')) return false;
298             queue.add(files, function () {
299                 //如果不存在正在上传的文件，且允许自动上传，上传该文件
300                 if (curId == EMPTY && autoUpload) {
301                     self.uploadFiles();
302                 }
303             });
304         },
305         /**
306          * 向上传按钮容器内增加用于存储文件路径的input
307          */
308         _renderUrlsInput:function () {
309             var self = this, button = self.get('button'), inputWrapper = button.target,
310                 name = self.get('urlsInputName'),
311                 urlsInput = new UrlsInput(inputWrapper, {name:name});
312             urlsInput.render();
313             return urlsInput;
314         },
315         /**
316          * 当上传完毕后返回结果集的处理
317          */
318         _uploadCompleteHanlder:function (ev) {
319             var self = this, result = ev.result, status, event = Uploader.event,
320                 queue = self.get('queue'), index = self.get('curUploadIndex');
321             if (!S.isObject(result)) return false;
322             //文件上传状态
323             status = result.status;
324             if (status) {
325                 //修改队列中文件的状态为success（上传完成）
326                 queue.fileStatus(index, Uploader.status.SUCCESS);
327                 self._success(result.data);
328                 self.fire(event.SUCCESS,{index : index,file : queue.getFile(index)});
329             } else {
330                 var msg = result.msg || EMPTY;
331                 //修改队列中文件的状态为error（上传失败）
332                 queue.fileStatus(index, Uploader.status.ERROR, msg);
333                 self.fire(event.ERROR, {status:status});
334             }
335             //置空当前上传的文件在队列中的索引值
336             self.set('curUploadIndex', EMPTY);
337             self.fire(event.COMPLETE,{index : index,file : queue.getFile(index)});
338             //存在批量上传操作，继续上传
339             self._continueUpload();
340         },
341         /**
342          * 取消上传后调用的方法
343          */
344         _uploadStopHanlder:function () {
345             var self = this, queue = self.get('queue'),
346                 index = self.get('curUploadIndex');
347             //更改取消上传后的状态
348             queue.fileStatus(index, Uploader.status.CANCEL);
349             //重置当前上传文件id
350             self.set('curUploadIndex', EMPTY);
351             self.fire(Uploader.event.CANCEL,{index : index});
352         },
353         /**
354          * 如果存在批量上传，则继续上传
355          */
356         _continueUpload : function(){
357             var self = this,
358                 uploadFilesStatus = self.get('uploadFilesStatus');
359             if (uploadFilesStatus != EMPTY) {
360                 self._uploaderStatusFile(uploadFilesStatus);
361             }
362         },
363         /**
364          * 上传进度监听器
365          */
366         _uploadProgressHandler:function (ev) {
367             var self = this, queue = self.get('queue'),
368                 index = self.get('curUploadIndex'),
369                 file = queue.getFile(index);
370             S.mix(ev,{file : file});
371             queue.fileStatus(index, Uploader.status.PROGRESS, ev);
372             self.fire(Uploader.event.PROGRESS,ev);
373         },
374         /**
375          * 上传成功后执行的回调函数
376          * @param {Object} data 服务器端返回的数据
377          */
378         _success:function (data) {
379             if (!S.isObject(data)) return false;
380             var self = this, url = data.url,
381                 urlsInput = self.get('urlsInput'),
382                 fileId = self.get('curUploadId'),
383                 queue = self.get('queue');
384             if (!S.isString(url) || !S.isObject(urlsInput)) return false;
385             //追加服务器端返回的文件url
386             queue.updateFile(fileId, {'sUrl':url});
387             //向路径隐藏域添加路径
388             urlsInput.add(url);
389         },
390         _error:function () {
391 
392         }
393     }, {ATTRS:/** @lends Uploader*/{
394         /**
395          * Button按钮的实例
396          */
397         button:{value:{}},
398         /**
399          * Queue队列的实例
400          */
401         queue:{value:{}},
402         /**
403          * 采用的上传方案，auto：根据浏览器自动选择，iframe：采用iframe方案，ajax：采用ajax方案
404          */
405         type:{value:Uploader.type.AUTO},
406         /**
407          * 服务器端配置
408          */
409         serverConfig:{value:{action:EMPTY, data:{}, dataType:'json'}},
410         /**
411          * 是否允许上传文件
412          */
413         isAllowUpload:{value:true},
414         /**
415          * 是否自动上传
416          */
417         autoUpload:{value:true},
418         /**
419          * 允许上传的文件最大大小，iframe上传方式不支持大小验证，务必服务器端也对文件大小进行验证
420          */
421         /*maxSize : {value : 5000},*/
422         /**
423          * 存储文件路径的隐藏域的name名
424          */
425         urlsInputName:{value:EMPTY},
426         //当前上传的文件对应的在数组内的索引值
427         curUploadIndex:{value:EMPTY},
428         uploadType:{value:{}},
429         urlsInput:{value:EMPTY},
430         //存在批量上传文件时，指定的文件状态
431         uploadFilesStatus:{value:EMPTY}
432     }});
433     return Uploader;
434 }, {requires:['base', 'node', './urlsInput', './type/iframe', './type/ajax', './type/flash', 'flash']});
