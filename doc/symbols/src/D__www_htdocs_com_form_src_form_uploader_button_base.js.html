1 /**
  2  * @fileoverview 文件上传按钮base
  3  * @author: 紫英(橘子)<daxingplay@gmail.com>, 剑平（明河）<minghe36@126.com>
  4  **/
  5 KISSY.add('form/uploader/button/base',function(S, Node, Base) {
  6     var EMPTY = '',
  7         LOG_PREFIX = '[AjaxUploader-Button] ',
  8         $ = Node.all;
  9 
 10     /**
 11      * 文件上传按钮
 12      * @class Button
 13      * @constructor
 14      * @param {Object} config 配置对象
 15      */
 16     function Button(target, config) {
 17         var self = this;
 18         //超类初始化
 19         Button.superclass.constructor.call(self, config);
 20         self.set('target', $(target));
 21     }
 22 
 23     S.mix(Button, {
 24         //支持的事件
 25         event : {
 26             'beforeShow': 'beforeShow',
 27             'afterShow': 'afterShow',
 28             'beforeHide': 'beforeHide',
 29             'afterHide': 'afterHide',
 30             'beforeRender' : 'beforeRender',
 31             'afterRender' : 'afterRender',
 32             'CHANGE' : 'change'
 33         }
 34     });
 35 
 36     S.extend(Button, Base, /** @lends Button.prototype*/{
 37         /**
 38          * 运行
 39          * @return {Object} Button的实例
 40          */
 41         render : function() {
 42             var self = this,
 43                 target = self.get('target'),
 44                 render = self.fire(Button.event.beforeRender);
 45             if (render === false) {
 46                 S.log(LOG_PREFIX + 'button render was prevented.')
 47                 return false;
 48             } else {
 49                 if (target == null) {
 50                     S.log(LOG_PREFIX + 'Cannot find target!');
 51                     return false;
 52                 }
 53                 self._createInput();
 54                 self.fire(Button.event.afterRender);
 55                 S.log(LOG_PREFIX + 'button was rendered just now.');
 56                 return self;
 57             }
 58         },
 59         /**
 60          * 显示按钮
 61          */
 62         show : function() {
 63             var self = this,
 64                 target = self.get('target'),
 65                 disableCls = self.get('cls').disabled,
 66                 input = self.get('fileInput'),
 67                 show = self.fire(Button.event.beforeShow);
 68             if (show === false) {
 69                 S.log(LOG_PREFIX + 'show button event was prevented.');
 70             } else {
 71                 // $(target).show();
 72                 $(target).removeClass(disableCls);
 73                 $(input).show();
 74                 self.fire(Button.event.afterShow);
 75                 S.log(LOG_PREFIX + 'button showed.');
 76             }
 77         },
 78         /**
 79          * 隐藏按钮
 80          */
 81         hide : function() {
 82             var self = this,
 83                 target = self.get('target'),
 84                 disableCls = self.get('cls').disabled,
 85                 input = self.get('fileInput'),
 86                 hide = self.fire(Button.event.beforeHide);
 87             if (hide === false) {
 88                 S.log(LOG_PREFIX + 'hide button event was prevented.');
 89             } else {
 90                 // $(target).hide();
 91                 $(target).addClass(disableCls);
 92                 $(input).hide();
 93                 self.fire(Button.event.afterHide);
 94                 S.log(LOG_PREFIX + 'button showed.');
 95             }
 96         },
 97         /**
 98          * 重置按钮
 99          * @return {Object} Button的实例
100          */
101         _reset : function() {
102             var self = this,
103                 inputContainer = self.get('inputContainer');
104             //移除表单上传域容器
105             $(inputContainer).remove();
106             self.set('inputContainer', EMPTY);
107             self.set('fileInput', EMPTY);
108             //重新创建表单上传域
109             self._createInput();
110             return self;
111         },
112         /**
113          * 创建隐藏的表单上传域
114          * @return {HTMLElement} 文件上传域容器
115          */
116         _createInput : function() {
117             var self = this,
118                 target = self.get('target'),
119                 name = self.get('name'),
120                 tpl = self.get('tpl'),
121                 multiple = self.get('multiple'),
122                 html,
123                 inputContainer,
124                 fileInput;
125             if (!S.isString(name) || !S.isString(tpl)) {
126                 S.log(LOG_PREFIX + 'No name or tpl specified.');
127                 return false;
128             }
129             html = S.substitute(tpl, {
130                 'name' : name
131             });
132             // TODO: inputContainer = DOM.create(html);
133             inputContainer = $(html);
134             //向body添加表单文件上传域
135             $(inputContainer).appendTo(target);
136             fileInput = $(inputContainer).children('input');
137             // TODO: 开启多选上传
138             // multiple && DOM.attr('multiple', 'multiple');
139             //上传框的值改变后触发
140             $(fileInput).on('change', self._changeHandler, self);
141             //DOM.hide(fileInput);
142             self.set('fileInput', fileInput);
143             self.set('inputContainer', inputContainer);
144             // self.resetContainerCss();
145             return inputContainer;
146         },
147         /**
148          * 文件上传域的值改变时触发
149          * @param {Object} ev 事件对象
150          */
151         _changeHandler : function(ev) {
152             var self = this,
153                 fileInput = self.get('fileInput'),
154                 value = $(fileInput).val();
155             if (value == EMPTY) {
156                 S.log(LOG_PREFIX + 'No file selected.');
157                 return false;
158             }
159             self.fire(Button.event.CHANGE, {
160                 files: ev.target.files,
161                 input: $(fileInput).clone().getDOMNode()
162             });
163             S.log(LOG_PREFIX + 'button change event was fired just now.');
164             // change完之后reset按钮，防止选择同一个文件无法触发change事件
165             self._reset();
166         }
167     }, {
168         ATTRS : /** @lends Button */{
169             /**
170              * target
171              */
172             target: {
173                 value: null
174             },
175             /**
176              * 对应的表单上传域
177              * @type HTMLElement
178              */
179             fileInput: {
180                 value: EMPTY
181             },
182             inputContainer: {
183                 value: EMPTY
184             },
185             /**
186              * 隐藏的表单上传域的模板
187              * @type String
188              */
189             tpl : {
190                 value : '<div class="ks-ajax-uploader-input-container"><input type="file" name="{name}" hidefoucs="true" class="ks-ajax-uploader-input" /></div>'
191             },
192             /**
193              * 隐藏的表单上传域的name值
194              * @type String
195              */
196             name : {
197                 value : 'fileInput',
198                 setter : function(v) {
199                     if (this.get('fileInput')) {
200                         $(this.get('fileInput')).attr('name', v);
201                     }
202                     return v;
203                 }
204             },
205             /**
206              * 是否可用,false为可用
207              * @type Boolean
208              */
209             disabled : {
210                 value : false,
211                 setter : function(v) {
212                     var self = this;
213                     if (v) {
214                         self.hide();
215                     } else {
216                         self.show();
217                     }
218                     return v;
219                 }
220             },
221             /**
222              * 样式
223              * @type Object
224              */
225             cls : {
226                 value : {
227                     disabled : 'uploader-button-disabled'
228                 }
229             }
230         }
231     });
232 
233     return Button;
234 
235 }, {
236     requires:[
237         'node',
238         'base'
239     ]
240 });
