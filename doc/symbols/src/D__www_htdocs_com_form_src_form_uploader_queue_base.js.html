1 /**
  2  * @fileoverview 文件上传队列列表显示和处理
  3  * @author 剑平（明河）<minghe36@126.com>,紫英<daxingplay@gmail.com>
  4  **/
  5 KISSY.add('form/uploader/queue/base',function(S, Node, Base, Status) {
  6     var EMPTY = '',$ = Node.all,LOG_PREFIX = '[uploader-queue]:';
  7 
  8     /**
  9      * @name Queue
 10      * @class 文件上传队列
 11      * @constructor
 12      * @extends Base
 13      * @requires Node,Status
 14      */
 15     function Queue(target, config) {
 16         var self = this;
 17         //调用父类构造函数
 18         Queue.superclass.constructor.call(self, config);
 19         //队列目标
 20         self.set('target', $(target));
 21     }
 22 
 23     S.mix(Queue, /**@lends Queue*/ {
 24         /**
 25          * 模板
 26          */
 27         tpl : {
 28             DEFAULT:'<li id="queue-file-{id}" class="clearfix" data-name="{name}">' +
 29                 '<div class="f-l sprite file-icon"></div>' +
 30                 '<div class="f-l">{name}</div>' +
 31                 '<div class="f-l file-status J_FileStatus"></div>' +
 32                 '</li>'
 33         },
 34         /**
 35          * 支持的事件
 36          */
 37         event : {
 38             //添加完一个文件后的事件
 39             ADD : 'add',
 40             //添加多个文件后的事件
 41             ADD_ALL : 'addAll',
 42             //删除文件后触发
 43             REMOVE : 'remove',
 44             // 队列满时触发
 45             QUEUE_FULL: 'queueFull'
 46         },
 47         /**
 48          * 文件的状态
 49          */
 50         status : Status.type,
 51         //样式
 52         cls : {
 53             QUEUE : 'ks-uploader-queue'
 54         },
 55         hook : {
 56             //状态
 57             STATUS : '.J_FileStatus'
 58         }
 59     });
 60     //继承于Base，属性getter和setter委托于Base处理
 61     S.extend(Queue, Base, /** @lends Queue.prototype*/{
 62         /**
 63          * 运行组件
 64          * @return {Queue}
 65          */
 66         render: function() {
 67             var self = this,$target = self.get('target');
 68             $target.addClass(Queue.cls.QUEUE);
 69             return self;
 70         },
 71         /**
 72          * 向上传队列添加文件
 73          * @param {Object} file 文件信息，格式类似{'name' : 'test.jpg','size' : 2000,'input' : {},'file' : {'name' : 'test.jpg','type' : 'image/jpeg','size' : 2000}}
 74          * @return {NodeList} 文件节点
 75          */
 76         add : function(file,callback) {
 77             var self = this,$target = self.get('target'),event = Queue.event,hFile,elFile,
 78                 //预置文件id
 79                 autoId = self.get('id'),
 80                 //文件信息显示模板
 81                 tpl = self.get('tpl'),
 82                 files = self.get('files'),
 83                 uploader = self.get('uploader'),
 84                 duration = self.get('duration');
 85             if(!S.isObject(file)){
 86                 S.log(LOG_PREFIX + 'add()参数file不合法！');
 87                 return false;
 88             }
 89             //设置文件唯一id
 90             file.id = autoId;
 91             //转换文件大小单位为（kb和mb）
 92             if(file.size) file.textSize = Status.convertByteSize(file.size);
 93             hFile = S.substitute(tpl, file);
 94             //将文件添加到队列之中
 95             elFile = $(hFile).hide().appendTo($target).data('data-file', file);
 96             //文件层
 97             file.target = elFile;
 98             //状态实例
 99             file.status = self._renderStatus(file);
100             if(S.isObject(uploader)) file.status.set('uploader',uploader);
101             files[autoId] = file;
102             self.set('files', files);
103             //设置文件状态为等待上传
104             self.fileStatus(autoId, Queue.status.WAITING);
105             //增加文件id编号
106             self.set('id', autoId + 1);
107             elFile.fadeIn(duration,function(){
108                 callback && callback.call(self,autoId,file);
109                 self.fire(event.ADD, {id : autoId,file : file,target : file.target});
110             });
111             return autoId;
112         },
113         /**
114          * 删除队列中指定id的文件
115          * @param {Number} id 文件id
116          * @param {Function} callback 删除元素后执行的回调函数
117          */
118         remove : function(id,callback){
119             var self = this,files = self.get('files'),file = files[id],$file,
120                 duration = self.get('duration');
121             if(S.isObject(file)){
122                 $file = file.target;
123                 $file.fadeOut(duration,function(){
124                     $file.remove();
125                     callback && callback.call(self,id);
126                     self.fire(Queue.event.REMOVE,{id : id,file : file});
127                 });
128                 delete files[id];
129                 self.set('files',files);
130             }
131             return file;
132         },
133         /**
134          * 获取或设置文件状态
135          * @param {Number} id 文件id
136          * @param {String} status 文件状态
137          * @return {Object}
138          */
139         fileStatus : function(id, status,args) {
140             if (!S.isNumber(id)) return false;
141             var self = this,files = self.get('files'),file = self.getFile(id),
142                 st = Queue.status,oStatus;
143             if (!S.isPlainObject(file)) return false;
144             //状态实例
145             oStatus = file['status'];
146             if(status) oStatus.change(status,args);
147             return  oStatus;
148         },
149         /**
150          * 获取指定索引值的队列中的文件
151          * @param  {Number} id 文件id
152          * @return {Object}
153          */
154         getFile : function(id) {
155             if (!S.isNumber(id)) return false;
156             var self = this,files = self.get('files'),
157                 file = files[id];
158             if (!S.isPlainObject(file)) file = false;
159             return file;
160         },
161         /**
162          * 更新文件数据对象，你可以追加数据
163          * @param {Number} id 文件id
164          * @param {Object} data 数据
165          * @return {Object}
166          */
167         updateFile : function(id,data){
168             if (!S.isNumber(id)) return false;
169             if(!S.isObject(data)){
170                 S.log(LOG_PREFIX + 'updateFile()的data参数有误！');
171                 return false;
172             }
173             var self = this,files = self.get('files'),
174                 file = self.getFile(id);
175             if(!file) return false;
176             S.mix(file,data);
177             files[id] = file;
178             self.set('files',files);
179             return file;
180         },
181         /**
182          * 获取等待状态的文件id数组
183          */
184         getWaitFileIds : function(){
185             var self = this,files = self.get('files'),
186                 status,waitFileIds = [];
187             if(!files.length) return waitFileIds;
188             S.each(files,function(file,index){
189                 if(S.isObject(file)){
190                     status = file.status;
191                     //文件状态
192                     if(status.get('curType') == status.constructor.type.WAITING){
193                         waitFileIds.push(index);
194                     }
195                 }
196             });
197             return waitFileIds;
198         },
199         /**
200          * 运行Status
201          * @param {Object} file  文件数据
202          * @return {Status} 状态实例
203          */
204         _renderStatus : function(file) {
205             var self = this,$file = file.target,hook = Queue.hook.STATUS,elStatus;
206             if (!$file.length) return false;
207             //状态层
208             elStatus = $file.children(hook);
209             //实例化状态类
210             return new Status(elStatus,{queue : self,file : file});
211         }
212     }, {ATTRS : /** @lends Queue*/{
213         /**
214          * 模板
215          * @type String
216          */
217         tpl : { value : Queue.tpl.DEFAULT },
218         /**
219          * 动画速度
220          */
221         duration : {value : 0.3},
222         target : {value : EMPTY},
223         id : {value : 0},
224         files : {value : []},
225         //上传组件实例
226         uploader : {value : EMPTY}
227     }});
228 
229     return Queue;
230 }, {requires:['node','base','./status']});
