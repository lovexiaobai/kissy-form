1 /**
  2  * 上传组件验证规则
  3  */
  4 KISSY.add(function(S, Node, Validation) {
  5     var Util = Validation.Util,Rule = Validation.Rule;
  6     //uploader为Uploader的实例
  7     Rule.add('uploader-max','最多上传{0}个文件！',function(value,text,max,uploader){
  8         var urlsInput = uploader.get('urlsInput'),
  9             len = urlsInput.get('urls').length;
 10         if(len > max){
 11             //不允许文件上传
 12             uploader.set('isAllowUpload',false);
 13             text = S.substitute(text,{"0" : max});
 14             return text;
 15         }
 16     });
 17     Rule.add('uploader-ext','不支持{0}格式文件上传！',function(value,text,exts,uploader){
 18          //队列
 19         var queue = uploader.get('queue'),
 20             curUploadId = queue.get('files').length - 1,
 21             //文件数据
 22             file = queue.getFile(curUploadId),
 23             //文件名
 24             fileName = file.name,
 25             //文件扩展名
 26             fileExt = _getFileExt(fileName),
 27             isAllow = _isAllowUpload(fileName);
 28         if(!isAllow){
 29             _stopUpload();
 30             return text;
 31         }
 32         /**
 33          * 是否允许上传
 34          * @param {String} fileName 文件名
 35          * @return {Boolean}
 36          */
 37         function _isAllowUpload(fileName){
 38             var isAllow = false,reg;
 39             S.each(exts, function(ext) {
 40                 reg = new RegExp('^.+\.' + ext + '$');
 41                 //存在该扩展名
 42                 if (reg.test(fileName))  return isAllow = true;
 43             });
 44             return isAllow;
 45         }
 46         /**
 47          * 阻止文件上传，并改变文件状态为error
 48          */
 49         function _stopUpload(){
 50             //不允许文件上传
 51             uploader.set('isAllowUpload',false);
 52             text = S.substitute(text,{"0" : fileExt});
 53             //改变文件状态为error
 54             queue.fileStatus(file.id,queue.constructor.status.ERROR,{msg : text});
 55         }
 56         /**
 57          * 获取文件扩展名
 58          * @param {String} file
 59          */
 60         function _getFileExt(file){
 61             var arr = file.split('.');
 62             return arr[arr.length -1];
 63         }
 64     });
 65     Rule.add('uploader-exist','该文件已经存在！',function(value,text,uploader){
 66         var queue = uploader.get('queue'),
 67             curUploadId = queue.get('files').length - 1,
 68             //文件数据
 69             file = queue.getFile(curUploadId),
 70             //文件名
 71             fileName = file.name;
 72         
 73     });
 74     Rule.add('uploader-require','必须至少上传一个文件！',function(value){
 75 
 76     });
 77     return Rule;
 78 }, {requires : ['node','validation']});
