  5 KISSY.add('form/uploader/render',function (S, Base, Node, Uploader, Button,SwfButton) {
  6     var EMPTY = '', $ = Node.all, LOG_PREFIX = '[uploaderRender]:',
  7         dataName = {
  8             CONFIG:'data-config',
  9             BUTTON_CONFIG : 'data-button-config',
 10             THEME_CONFIG : 'data-theme-config'
 11         };
 12 
 13     /**
 14      * 解析组件在页面中data-config成为组件的配置
 15      * @param {String} hook 组件钩子
 16      * @param {String} dataConfigName 配置名
 17      * @return {Object}
 18      */
 19     S.parseConfig = function(hook, dataConfigName) {
 20         var config = {}, sConfig, DATA_CONFIG = dataConfigName || dataName.CONFIG;
 21         sConfig = $(hook).attr(DATA_CONFIG);
 22         if (!S.isString(sConfig)) return {};
 23         try {
 24             config = JSON.parse(sConfig);
 25         } catch (err) {
 26             S.log(LOG_PREFIX + '请检查' + hook + '上' + DATA_CONFIG + '属性内的json格式是否符合规范！');
 27         }
 28         return config;
 29     };
 30 
 31     /**
 32      * @name RenderUploader
 33      * @class 运行文件上传组件
 34      * @constructor
 35      * @param {String | HTMLElement} buttonTarget 上传按钮目标元素
 36      * @param {String | HTMLElement} queueTarget 文件队列目标元素
 37      * @param {Object} config 配置
 38      */
 39     function RenderUploader(buttonTarget, queueTarget, config) {
 40         var self = this;
 41         //合并配置
 42         config = S.mix(S.parseConfig(buttonTarget), config);
 43         //超类初始化
 44         RenderUploader.superclass.constructor.call(self, config);
 45         self.set('buttonTarget', buttonTarget);
 46         self.set('queueTarget', queueTarget);
 47         self.set('uploaderConfig', config);
 48         self._init();
 49     }
 50 
 51     S.extend(RenderUploader, Base, {
 52         /**
 53          * 初始化组件
 54          */
 55         _init:function () {
 56             var self = this, uploaderConfig = self.get('uploaderConfig'),
 57                 button = self._initButton(),
 58                 queue;
 59             self.set('button', button);
 60             self._initThemes(function (theme) {
 61                 queue = theme.get('queue');
 62                 //配置增加按钮实例和队列实例
 63                 S.mix(uploaderConfig, {button:button, queue:queue});
 64                 var uploader = new Uploader(uploaderConfig);
 65                 uploader.render();
 66                 self.set('uploader', uploader);
 67                 if(theme.afterUploaderRender) theme.afterUploaderRender(uploader);
 68                 self.fire('init', {uploader:uploader});
 69             });
 70         },
 71         /**
 72          * 初始化模拟的上传按钮
 73          * @return {Button}
 74          */
 75         _initButton:function () {
 76             var self = this,
 77                 target = self.get('buttonTarget'),
 78                 //从html标签的伪属性中抓取配置
 79                 config = S.parseConfig(target,dataName.BUTTON_CONFIG),
 80                 name = self.get('name'),
 81                 type = self.get('type');
 82             //合并配置
 83             config = S.merge({name:name},config);
 84             //实例化上传按钮
 85             return type != 'flash' && new Button(target, config) || new SwfButton(target);
 86         },
 87         _initThemes:function (callback) {
 88             var self = this, theme = self.get('theme'),
 89                 target = self.get('buttonTarget'),
 90                 //从html标签的伪属性中抓取配置
 91                 config = S.parseConfig(target,dataName.THEME_CONFIG);
 92             S.use(theme + '/index', function (S, Theme) {
 93                 var queueTarget = self.get('queueTarget'),
 94                     theme;
 95                 S.mix(config,{queueTarget:queueTarget});
 96                 theme = new Theme(config);
 97                 callback && callback.call(self, theme);
 98             })
 99         },
100         /**
101          * 文件上传验证
102          */
103         _auth:function () {
104             /*var self = this,buttonTarget = self.get('buttonTarget'),
105              $btn = $(buttonTarget),

107              button = self.get('button'),
108              //TODO:需要修改
109              fileInput = button.fileInput,
110              DATA_NAME = dataName.VALID, valid;
111              if(!$btn.length) return false;
112              valid = $btn.attr(DATA_NAME);
113              //不存在验证配置，直接退出
114              if(!valid) return false;
115              $(fileInput).attr(DATA_NAME,valid);*/
116         }
117     }, {
118         ATTRS:{
119             theme:{value:'form/uploader/themes/default' },
120             /**
121              * 按钮目标元素
122              */
123             buttonTarget:{value:EMPTY},
124             /**
125              * 队列目标元素
126              */
127             queueTarget:{value:EMPTY},
128             /**
129              * 上传组件配置
130              */
131             uploaderConfig:{},
132             /**
133              * Button（上传按钮）的实例
134              */
135             button:{value:EMPTY},
136             /**
137              * Queue（上传队列）的实例
138              */
139             queue:{value:EMPTY},
140             uploader:{value:EMPTY}
141         }
142     });
143     return RenderUploader;
144 }, {requires:['base', 'node', './base', './button/base','./button/swfButton']});
