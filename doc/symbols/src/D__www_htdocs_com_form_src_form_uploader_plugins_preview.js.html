153                         Math.max( 0, self.config.maxWidth ) / width  || 1,Math.max( 0, self.config.maxHeight ) / height || 1
154                     );
155 					self.img.style.width = Math.round( width * ratio ) + "px";
156                		self.img.style.height = Math.round( height * ratio ) + "px";
157                		self.img.setAttribute('data-ratio', ratio);
158 				}
159 				// for ImageZoom

161 				D.attr(self.img, 'data-ks-imagezoom', imagezoomSrc);
162 				// self.config.onShow();
163 				self.fire(self.event.show);
164 			}
